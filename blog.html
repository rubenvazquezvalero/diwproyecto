'use strict';

let frase = prompt("");

function esPalindroma(palabra){
    palabra = palabra.toLocaleLowerCase().replace(/ /g, "");
    let resul = true;
    let fin = palabra.length - 1;
    for(let i = 0 ; i<fin && resul; i++, fin--){
        resul = palabra[i] == palabra[fin];
    }
    return resul;
}

let resultado = esPalindroma(frase);

document.write(resultado ? "\""+frase+"\" es palíndroma.": "\""+frase+"\" no es palíndroma.");
